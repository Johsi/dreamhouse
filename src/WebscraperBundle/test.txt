			}

				$("img:last", currentGallery).show();
			}

			.addClass("hintText")
			.val(hintText)

				if ($(this).val() == hintText) {
					$(this)
						.val("")
						.removeClass("hintText");
				}
			})
			.focusout(function () {
				if ($(this).val() == "") {
					$(this)

			if ($(this).text() == "Select All") {

				$(".contactAllList li").addClass("selected");
			} else {
				$(this).text("Select All");
				$("input[type=checkbox]", ".contactAllList").attr("checked", false);
				$(".contactAllList li").removeClass("selected");
			}

		});

		$("#contactAllSubmit").live("click", function (e) {

			if ($(".contactAllTextarea").hasClass("hintText")) {
				$(".contactAllTextarea").val("").removeClass("hintText");
			}

			$('#contactAllForm').validate({
				rules: {
					Email: { email: true },
					Phone: { phone: true }
				},
				messages: {
					Name: "Please enter your name",
					Email: "Please enter your email address",
					Phone: "Please enter your phone number",
					Message: "Please enter a message",
					UserType: "Please select one"
				},
				showErrors: function (errorMap, errorList) {
					this.defaultShowErrors();
					$('#contactAllErrorMessages').hide();
				},
				errorPlacement: function (error, element) {
					error.insertBefore(element);
				}
			});

			var groupIDs = new Array();
			$("input:checked").each(function (i) {
				var groupID = parseInt($(this).attr("name"));
				if (!isNaN(groupID))
					groupIDs.push(groupID);
			});
			var agents = groupIDs.join(',');

			var isAgentsValid = true;
			if (agents == "" || agents == null) {
				$("#contactAllListErrors").html("<span>You have not selected any agents. Please select at least one agent from the list</span>");
				isAgentsValid = false;
			} else if (groupIDs.length > 30) {
				$("#contactAllListErrors").html("<span>You've selected " + groupIDs.length + " agents. Please select maximum 30 agents from the list</span>");
				isAgentsValid = false;
			} else {
				$("#contactAllListErrors").html("");
			}

			if ($('#contactAllForm').valid()) {
				if (!isAgentsValid) { return false; }

				$("#contactAllSubmit").text("Sending ...");
				$("#contactAllSubmit").attr("disabled", "disabled");
				$("#Agents").val(agents);

					url: $("#contactAllForm").attr('action'),
					data: $("#contactAllForm").serialize(),
					dataType: 'html',
					success: function (data) {
						$('#contactAllPanel').html(data);
					},
					error: function () { alert('An error occurred'); }
				});
			}
			return false;
		});

		if ($("#contactAllPanel").exists()) {
